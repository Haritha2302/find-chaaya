<script>
// ============================================================
// CONFIGURATION
// Paste the Apps Script Web App URL below:
// ============================================================
const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbwo1VyT3C4QKiW1absjR5BdjTh7I4lW2q_tMii3IlSSh42cPotMNBl6z4DaBAu0gEIR/exec';

// ============================================================

// HOW THE MARKER IMAGE WORKS

//

// Use the chai glass PNG as the map marker.

// Leaflet's L.icon() defines a custom image as a

// marker. iconSize sets the display size on the map.

// iconAnchor is the pixel point on the icon that corresponds

// to the exact map location — [width/2, height] means the

// tip at the bottom-centre of the image is the anchor point.

// ============================================================

const chaiIcon = L.icon({

  iconUrl: 'marker.png',

  iconSize: [40, 48],

  iconAnchor: [20, 48],

  popupAnchor: [0, -48]

});


// Same icon but slightly larger for the selected state

const chaiIconSelected = L.icon({

  iconUrl: 'marker.png',

  iconSize: [50, 60],

  iconAnchor: [25, 60],

  popupAnchor: [0, -60]

});


// ============================================================

// MAP INITIALISATION

//

// L.map('map') creates a Leaflet map inside the <div id="map">

// setView([lat, lng], zoom) sets the starting position.

// Zoom 12 = city level, 6 = country level.

// Start centred on Kochi since all current shops are there.

//

// L.tileLayer() loads the map background tiles from

// OpenStreetMap — the actual street map imagery.

// {s} = subdomain, {z}/{x}/{y} = tile coordinates.

// ============================================================

const map = L.map('map', { zoomControl: true }).setView([9.98, 76.3], 12);


L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

  maxZoom: 19,

  attribution: '© OpenStreetMap contributors'

}).addTo(map);


// ============================================================

// STATE VARIABLES

//

// These variables live in memory and track the current state

// of the app — which shops are loaded, what's selected, etc.

// ============================================================

let allShops = [];           // all shops fetched from the sheet

let markers = [];            // Leaflet marker objects on the map

let selectedMarker = null;   // currently selected marker

let userMarker = null;       // the blue dot showing your location

let routeLine = null;        // the dashed route line on the map

let activeShop = null;       // the shop currently shown in the panel


// ============================================================

// FETCHING SHOP DATA

//

// fetch() is the browser's built-in function to make HTTP

// requests. We call the Apps Script URL, which returns JSON.

// .then(r => r.json()) parses the JSON text into a JS object.

// Once we have the data, we call renderMarkers() to plot them.

// ============================================================

fetch(SCRIPT_URL)

  .then(r => r.json())

  .then(data => {

    allShops = data.shops;

    renderMarkers(allShops);

    hideLoading();

  })

  .catch(err => {

    // If the fetch fails (wrong URL, network issue), show an

    // error so the user knows something went wrong.

    console.error('Failed to load shop data:', err);

    document.querySelector('#loading p').textContent = 'Could not load shops. Check the script URL.';

  });


// ============================================================

// RENDERING MARKERS

//

// This function takes an array of shops and places a chai

// glass marker on the map for each one.

//

// First we remove all existing markers (in case we're

// re-rendering after a filter change), then add fresh ones.

//

// marker.on('click', ...) attaches a click handler so tapping

// a glass opens the info panel for that shop.

// ============================================================

function renderMarkers(shops) {

  // Remove old markers from the map

  markers.forEach(m => map.removeLayer(m));

  markers = [];


  shops.forEach(shop => {

    // Skip shops with missing coordinates

    if (!shop.lat || !shop.lng) return;


    const marker = L.marker([shop.lat, shop.lng], { icon: chaiIcon }).addTo(map);


    marker.on('click', () => {

      selectShop(shop, marker);

    });


    markers.push(marker);

  });


  // Update the count label in the filter bar

  document.getElementById('shop-count').textContent =

    shops.length === 1 ? '1 shop' : shops.length + ' shops';

}


// ============================================================

// SELECTING A SHOP

//

// When you tap a marker, this function:

// 1. Resets the previously selected marker back to normal size

// 2. Makes the tapped marker larger (selected state)

// 3. Populates the info panel with that shop's details

// 4. Slides the panel up into view

// ============================================================

function selectShop(shop, marker) {

  // Reset previous selection

  if (selectedMarker) selectedMarker.setIcon(chaiIcon);


  // Highlight new selection

  marker.setIcon(chaiIconSelected);

  selectedMarker = marker;

  activeShop = shop;


  // Fill in the panel

  document.getElementById('panel-name').textContent = shop.name;

  document.getElementById('panel-location').textContent =

    [shop.area, shop.city, shop.state].filter(Boolean).join(', ');


  document.getElementById('taste-stars').innerHTML = starsFor(shop.taste);

  document.getElementById('taste-num').textContent = shop.taste + ' / 5';

  document.getElementById('vibe-stars').innerHTML = starsFor(shop.vibe);

  document.getElementById('vibe-num').textContent = shop.vibe + ' / 5';


  // Set the Google Maps link

  const gmapsBtn = document.getElementById('gmaps-btn');

  if (shop.gmaps) {

    gmapsBtn.href = shop.gmaps;

    gmapsBtn.style.display = 'block';

  } else {

    gmapsBtn.style.display = 'none';

  }


  // Hide any previous route info

  document.getElementById('route-row').classList.remove('visible');


  // Open the panel by adding the 'open' class (see CSS)

  document.getElementById('info-panel').classList.add('open');


  // Pan the map to centre on the selected shop

  map.panTo([shop.lat, shop.lng]);

}


// ============================================================

// STAR RATINGS

//

// Converts a number (e.g. 4) into filled/empty star characters

// e.g. starsFor(4) → "★★★★☆"

// Math.round handles half-values like 3.5 → 4

// ============================================================

function starsFor(n) {

  const val = Number(n);

  let html = '';

  for (let i = 1; i <= 5; i++) {

    if (val >= i) {

      html += '<span style="color:#C8874A">★</span>';

    } else if (val >= i - 0.5) {

      html += '<span style="background:linear-gradient(to right,#C8874A 50%,#ddd 50%);-webkit-background-clip:text;-webkit-text-fill-color:transparent;background-clip:text;">★</span>';

    } else {

      html += '<span style="color:#ddd">★</span>';

    }

  }

  return html;

}


// ============================================================

// CLOSING THE PANEL

// ============================================================

function closePanel() {

  document.getElementById('info-panel').classList.remove('open');

  if (selectedMarker) selectedMarker.setIcon(chaiIcon);

  selectedMarker = null;

  activeShop = null;

  if (routeLine) { map.removeLayer(routeLine); routeLine = null; }

}


// Close panel when tapping the map background

map.on('click', closePanel);


// ============================================================

// FILTERING

//

// When either slider moves, re-filter allShops to only

// those where taste >= tasteMin AND vibe >= vibeMin,

// then re-render the markers.

//

// allShops never changes — it always holds the full list.

// This just passes a filtered subset to renderMarkers().

// ============================================================

const tasteSlider = document.getElementById('taste-filter');

const vibeSlider = document.getElementById('vibe-filter');


tasteSlider.addEventListener('input', applyFilters);

vibeSlider.addEventListener('input', applyFilters);


function applyFilters() {

  const tasteMin = Number(tasteSlider.value);

  const vibeMin = Number(vibeSlider.value);


  document.getElementById('taste-val').textContent = tasteMin;

  document.getElementById('vibe-val').textContent = vibeMin;


  const filtered = allShops.filter(s =>

    Number(s.taste) >= tasteMin && Number(s.vibe) >= vibeMin

  );


  closePanel();

  renderMarkers(filtered);

}


// ============================================================

// GPS — FIND ME

//

// navigator.geolocation is the browser API for GPS.

// getCurrentPosition() asks the user for permission and

// then returns their coordinates.

//

// We place a blue circle marker at their location.

// L.circleMarker draws a coloured circle (not an image icon).

// ============================================================

function locateMe() {

  if (!navigator.geolocation) {

    alert('Location not supported in this browser.');

    return;

  }

  navigator.geolocation.getCurrentPosition(pos => {

    const { latitude: lat, longitude: lng } = pos.coords;


    // Remove old user marker if it exists

    if (userMarker) map.removeLayer(userMarker);


    userMarker = L.circleMarker([lat, lng], {

      radius: 9,

      fillColor: '#4A90D9',

      fillOpacity: 1,

      color: 'white',

      weight: 2.5

    }).addTo(map);


    map.setView([lat, lng], 14);

  }, () => {

    alert('Could not get your location. Please allow location access.');

  });

}


// ============================================================

// ROUTE — GET WALKING ROUTE

//

// Call the OSRM public routing API (free, no key needed).

// It returns a GeoJSON LineString of the walking path between

// your location and the selected shop.

//

// L.geoJSON() draws that path as a dashed line on the map.

// Also calculate distance and walking time and show them

// in the route row inside the info panel.

//

// Walking speed assumed: 5 km/h (standard estimate)

// ============================================================

async function getRoute() {

  if (!activeShop) return;


  if (!userMarker) {

    alert('Tap ⊙ first to find your location.');

    return;

  }


  const userLatLng = userMarker.getLatLng();

  const url = `https://router.project-osrm.org/route/v1/foot/` +

    `${userLatLng.lng},${userLatLng.lat};${activeShop.lng},${activeShop.lat}` +

    `?overview=full&geometries=geojson`;


  try {

    const res = await fetch(url);

    const data = await res.json();

    const route = data.routes[0];


    // Distance comes back in metres — convert to km

    const distKm = (route.distance / 1000).toFixed(1);

    // Duration comes back in seconds — convert to minutes

    const mins = Math.round(route.duration / 60);

    const timeStr = mins >= 60

      ? `${Math.floor(mins / 60)}h ${mins % 60}m`

      : `${mins} min`;


    document.getElementById('route-dist').textContent = distKm + ' km';

    document.getElementById('route-time').textContent = timeStr;

    document.getElementById('route-row').classList.add('visible');


    // Draw the route on the map

    if (routeLine) map.removeLayer(routeLine);

    routeLine = L.geoJSON(route.geometry, {

      style: { color: '#4A90D9', weight: 3, opacity: 0.85, dashArray: '6, 5' }

    }).addTo(map);


    // Fit the map to show both your location and the shop

    const coords = route.geometry.coordinates;

    const bounds = L.latLngBounds(coords.map(c => [c[1], c[0]]));

    map.fitBounds(bounds, { padding: [60, 60] });


  } catch (e) {

    alert('Could not fetch route. Check your internet connection.');

  }

}


// ============================================================

// HIDE LOADING SCREEN

//

// Once data is loaded fade out the loading overlay.

// setTimeout with 0ms defers execution to after the current

// paint cycle so the map is visible before the

// overlay is removed — prevents a flash of unstyled content.

// ============================================================

function hideLoading() {

  setTimeout(() => {

    const el = document.getElementById('loading');

    el.style.opacity = '0';

    setTimeout(() => el.style.display = 'none', 400);

  }, 300);

}
